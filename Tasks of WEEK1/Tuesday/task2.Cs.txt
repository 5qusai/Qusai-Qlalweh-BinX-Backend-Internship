Fun(s1);

static void Fun(IPerson p)
{
    p.getInfo();
}

interface IPerson
{

    void getInfo();


}

public class Student : IPerson
{
    public string name;
    public Student(string name)
    {
        this.name = name;
    }

    public void getInfo()
    {
        Console.WriteLine($"Name:{name}");
    }

}

public class Instructor : IPerson
{
    public string name;
   public Instructor(string name)
    {
        this.name = name;
    }

    public void getInfo()
    {
        Console.WriteLine($"Name:{name}");
    }

}
